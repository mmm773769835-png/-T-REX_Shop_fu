# أوامر تشغيل تطبيق T-REX Shop
# نسخ هذه الأوامر وتشغيلها بالترتيب

# الخطوة 1: إيقاف جميع العمليات القديمة
taskkill /F /IM node.exe

# الخطوة 2: تشغيل السيرفر (افتح نافذة PowerShell جديدة)
cd server
node index.js

# الخطوة 3: تشغيل تطبيق Expo (افتح نافذة PowerShell جديدة)
cd ..
npx expo start --web

# الخطوة 4: فتح المتصفح
# افتح المتصفح على: http://localhost:8081
